void showbits( unsigned char n ){
    int i;
    unsigned char j,k, andmask;
    for(i=7;i>=0;i--){
        j=i;
        andmask = 1<<j;
        k = n&andmask;
        k==0?printf("0"):printf("1");
    }
}
