#include <stdio.h>
int display1(int);
int display2(int *);
int main() {
    int i;
    int marks[]={10,20,30,40,50};
    for(i=0;i<=4;i++)
        display1(marks[i]);
    for(i=0;i<=4;i++)
        display2(&marks[i]);
    return 0;
}
int display1(int m){
    printf("\n%d ", m);
}
int display2(int * n){
    printf("\n%d ", *n);
}
