#include <stdio.h>
int main() {
    int a[5][5]={
        {11,12,45,78,99},
        {5,4,9,8,7},
        {12,45,86,36,89},
        {55,44,66,22,88},
        {77,99,100,32,68}
    };
    int i,j,big;
    big=a[0][0];
    for(i=0;i<=4;i++){
        for(j=0;j<=4;j++){
            if(a[i][j]>big)
                big=a[i][j];
        }
    }
    printf("\nLargest number is %d\n", big);
   return 0;
}




Largest number is 100
