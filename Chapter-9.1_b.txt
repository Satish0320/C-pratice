#include <stdio.h>
int circularShift(int *x, int *y, int *z);
int main() {
    int a = 5, b = 8, c = 10;
    printf("Before circular shift:\n");
    printf("a = %d, b = %d, c = %d\n", a, b, c);
    circularShift(&a, &b, &c);
    printf("After circular shift:\n");
    printf("a = %d, b = %d, c = %d\n", a, b, c);
    return 0;
}
int circularShift(int *x, int *y, int *z) {
    int temp = *z;  
    *z = *y;       
    *y = *x;        
    *x = temp;    
}



Before circular shift:
a = 5, b = 8, c = 10
After circular shift:
a = 10, b = 5, c = 8
