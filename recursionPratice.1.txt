#include <stdio.h>
int fact(int );
int main() {
    int a, fac;
    printf("Enter the number: ");
    scanf("%d", &a);
    fac = fact(a);
    printf("The factorial is :%d\n",fac);
    return 0;
}

int fact(int x){
    int f;
    if(x==1)
        return(1);
    else
        f=x*fact(x-1);
        return(f);
}
