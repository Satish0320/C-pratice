#include <stdio.h>
int main() {
    int avg,sum=0,i;
    int marks[5];
    for(i=0;i<=4;i++){
        printf("Enter the marks: ");
        scanf("%d", &marks[i]);
    }
    for(i=0;i<=4;i++)
        sum=sum+marks[i];
        avg = sum/5;
        printf("The average is: %d", avg);
        return 0;
    
    return 0;
}
